SQL syntax for stored procedure used to join `ni_ptmarec` and ` ni_ptorderud`, and show the patientID, medicine name, dose, execute time at the same time.

DELIMITER //
CREATE PROCEDURE med(IN in_pt_no VARCHAR(8) charset utf8)
BEGIN
SELECT ni_ptorderud.PT_NO, ni_ptmarec.ptEncounterID, ni_ptmarec.OrderID, ni_ptmarec.MedPRS, ni_ptorderud.dose, ni_ptorderud.doseUnit, ni_ptmarec.exeDt FROM ni_PtOrderUD RIGHT JOIN ni_PtMARec
ON ni_ptorderud.PT_NO = in_pt_no AND ni_ptorderud.ptEncounterID = ni_ptmarec.ptEncounterID AND ni_ptorderud.OrderID = ni_ptmarec.OrderID;

END //
DELIMITER ;
